#include <stdio.h>
#include <iostream>
#include <string>
using namespace std;

string buf[50];
int main()
{
	int T;
	scanf("%d",&T);
	while(T--)
	{
		int n,m;
		scanf("%d%d",&n,&m);
		for(int i=0;i<n;++i)
				cin >> buf[i];
		int angle;
		scanf("%d",&angle);
		if(angle==0)//0度
		{
			for(int i=0;i<n;++i)
			{
				for(int j=0;j<m;j++)
				{
					printf("%c",buf[i][j]);	
				}
				printf("\n");
			}
		}
		else if(angle==90)//90度
		{
			for(int j=0;j<m;++j)
			{
				for(int i=n-1;i>=0;--i)
				{
					printf("%c",buf[i][j]);	
				}
				printf("\n");
			}
		}
		else if(angle==180)//180度
		{
		for(int i=n-1;i>=0;--i)
		{
			for(int j=m-1;j>=0;--j)
			{
				printf("%c",buf[i][j]);	
			}	
			printf("\n");
			}
		}
		else if(angle==270)//270度
		{
		for(int j=m-1;j>=0;--j)
		{
			for(int i=0;i<=n-1;++i)
			{
				printf("%c",buf[i][j]);	
			}
			printf("\n");
			}
		}
	}
	return 0;
}
