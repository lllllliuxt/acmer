# 使用两个vector来表示树结构
```c++
#include <bits/stdc++.h>
using namespace std;

bool mark[101];//判断进程是否存在
vector<vector<int> > PID(101);


void del(int id)//递归删除
{
	if(mark[id]==true)
	{
		mark[id]=false;
		for(int i=0;i<PID[id].size();++i)
		{
			del(PID[id][i]);
		}
	}
}

int main()
{
	int T;
	scanf("%d",&T);
	while(T--)
	{
		fill(mark,mark+101,false);
		for(int i=0;i<PID.size();++i)
		{
			PID[i].clear();
		}//初始化
		mark[0]=true;
		int N;
		scanf("%d",&N);
		while(N--)
		{
			char a[10];
			scanf("%s",a);
			string str=a;
			if(str=="FORK")
			{
				int PID1,PID2;
				scanf("%d%d",&PID1,&PID2);
				PID[PID1].push_back(PID2);
				mark[PID2]=true;
			}
			else if(str=="QUERY")
			{
				int PID1;
				scanf("%d",&PID1);
				printf("%s\n",mark[PID1]?"Yes":"No");
			}
			else if (str=="KILL")
			{
				int PID1;
				scanf("%d",&PID1);
				del(PID1);
			}
		}
	}
	return 0;
}
```
# 使用结构体来表示树结构

```c++
#include <bits/stdc++.h>

using namespace std;
struct Node{
	vector<int> son;
	bool flag; 
}Tree[101];

void creat(int index)
{
	Tree[index].son.clear();
	Tree[index].flag=true;
}

void del(int index)
{
	if(Tree[index].flag)
	{
		Tree[index].flag=false;
		for(int i=0;i<Tree[index].son.size();++i)
		{
			del(Tree[index].son[i]);
		}
	}
}

int main()
{
	int T;
	cin >> T;
	while(T--)
	{
		//初始化
		for(int i=0;i<101;++i)
		{
			Tree[i].son.clear();
			Tree[i].flag=false;
		 } 
		 
		int N;
		cin >> N;
		creat(0);
		while(N--)
		{
			string str;
			cin >> str;
			if(str=="FORK")
			{
				int pid1,pid2;
				cin >> pid1 >> pid2;
				Tree[pid1].son.push_back(pid2);
				creat(pid2);
			}
			else if(str=="QUERY")
			{
				int pid1;
				cin >> pid1;
				if(Tree[pid1].flag)
					cout << "Yes" << endl;
				else
					cout << "No" << endl;
			}
			else if(str=="KILL")
			{
				int pid1;
				cin >> pid1;
				del(pid1);
			}	
		}
	}
	return 0;
}
```
