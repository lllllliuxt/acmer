```c++
#include <bits/stdc++.h>
using namespace std;

bool mark[101];//判断进程是否存在
vector<vector<int> > PID(101);


void del(int id)//递归删除
{
	if(mark[id]==true)
	{
		mark[id]=false;
		for(int i=0;i<PID[id].size();++i)
		{
			del(PID[id][i]);
		}
	}
}

int main()
{
	int T;
	scanf("%d",&T);
	while(T--)
	{
		fill(mark,mark+101,false);
		for(int i=0;i<PID.size();++i)
		{
			PID[i].clear();
		}//初始化
		mark[0]=true;
		int N;
		scanf("%d",&N);
		while(N--)
		{
			char a[10];
			scanf("%s",a);
			string str=a;
			if(str=="FORK")
			{
				int PID1,PID2;
				scanf("%d%d",&PID1,&PID2);
				PID[PID1].push_back(PID2);
				mark[PID2]=true;
			}
			else if(str=="QUERY")
			{
				int PID1;
				scanf("%d",&PID1);
				printf("%s\n",mark[PID1]?"Yes":"No");
			}
			else if (str=="KILL")
			{
				int PID1;
				scanf("%d",&PID1);
				del(PID1);
			}
		}
	}
	return 0;
}
```
