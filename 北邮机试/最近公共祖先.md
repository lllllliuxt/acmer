```
#include <bits/stdc++.h>

using namespace std;
int parents[100];
int address1[100];
int address2[100];
int main()
{
	int T;
	scanf("%d",&T);
	parents[1]=-1;
	while(T--)
	{	
		int N;
		scanf("%d",&N);
		N--;
		while(N--)
		{
			int a,b;
			scanf("%d%d",&a,&b);
			parents[b]=a;
		}//输入 
				
		int M;
		scanf("%d",&M);
		while(M--)
		{
			int a,b;
			scanf("%d%d",&a,&b);
			int i=0,j=0;
			do
			{
				address1[i++]=a;
				a=parents[a];
			}while(a!=-1);//记录a到根的路径 
			do
			{
				address2[j++]=b;
				b=parents[b];
			}while(b!=-1);//记录b到根的路径 
			while(i!=0&&j!=0&&address1[i-1]==address2[j-1])
			{
				--i;
				--j;
			}
			printf("%d\n",address1[i]);
		}
	}
	return 0;
}
```
